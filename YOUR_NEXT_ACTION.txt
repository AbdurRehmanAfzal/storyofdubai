⚡ QUICK START (5 commands to see it working)

# Terminal 1: Database
docker run --name storyofdubai-postgres -e POSTGRES_PASSWORD=postgres \
  -p 5432:5432 -d postgres:16-alpine
docker run --name storyofdubai-redis -p 6379:6379 -d redis:7-alpine

# Terminal 2: Backend
cd backend && source venv/bin/activate && \
  cat > .env << 'EOF2'
ENVIRONMENT=development
DATABASE_URL=postgresql+asyncpg://postgres:postgres@localhost:5432/storyofdubai
REDIS_URL=redis://localhost:6379/0
EOF2
uvicorn app.main:app --reload --port 8000

# Terminal 3: Frontend
cd frontend && npm install && npm run dev

# Open browser: http://localhost:3000 ✅
