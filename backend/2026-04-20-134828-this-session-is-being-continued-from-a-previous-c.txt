  from typing import Any, Generic, Optional, TypeVar                            
  from pydantic import BaseModel                                                
                                                                                
  T = TypeVar("T")                                                              
                                                                                
  class PaginationMeta(BaseModel):                                              
      total: int                                                                
      page: int                                                                 
      per_page: int                                                             
      has_next: bool                                                            
      has_prev: bool                                                            
                                                                                
  class ErrorDetail(BaseModel):                                                 
      code: str                                                                 
      message: str                                                              
      details: dict = {}                                                        
                                                                                
  class APIResponse(BaseModel, Generic[T]):                                     
      success: bool                                                             
      data: Optional[T] = None                                                  
      meta: Optional[PaginationMeta] = None                                     
      error: Optional[ErrorDetail] = None                                       
                                                                                
      @classmethod                                                              
      def ok(cls, data: T, meta: PaginationMeta = None):                        
          return cls(success=True, data=data, meta=meta, error=None)            
                                                                                
      @classmethod                                                              
      def fail(cls, code: str, message: str, details: dict = {}):               
          return cls(success=False, data=None, meta=None,                       
                     error=ErrorDetail(code=code, message=message,              
  details=details))

  import math                                                                   
  from dataclasses import dataclass                                             
  from typing import Optional                                                   
  from app.scoring.base import BaseScorer, ScoreResult                          
                                                                                
  @dataclass                                                                    
  class VenueScoreInput:                                                        
      id: str                                                                   
      google_rating: Optional[float]                                            
      review_count: int                                                         
      price_tier: Optional[int]                                                 
      days_since_last_review: int                                               
      has_photos: bool                                                          
      has_phone: bool                                                           
      has_website: bool                                                         
      response_rate: Optional[float] = None                                     
                                                                                
  GLOBAL_MEAN_RATING = 3.8                                                      
  TRUST_REVIEW_COUNT = 50  # full trust at 50+ reviews                          
                                                                                
  WEIGHTS = {                                                                   
      "rating_quality": 30,                                                     
      "review_volume": 20,                                                      
      "recency": 20,                                                            
      "price_value": 15,                                                        
      "completeness": 15,                                                       
  }                                                                             
                                                                                
  class VenueScorer(BaseScorer):                                                
      def score(self, entity: VenueScoreInput) -> ScoreResult:                  
          breakdown = {}                                                        
                                                                                
          # 1. Rating quality (30 pts) — Bayesian average                       
          if entity.google_rating and entity.review_count > 0:                  
              weight = min(entity.review_count / TRUST_REVIEW_COUNT, 1.0)       
              bayesian = (weight * entity.google_rating) + ((1 - weight) *      
  GLOBAL_MEAN_RATING)                                                           
              pts = (bayesian / 5.0) * WEIGHTS["rating_quality"]                
          else:                                                                 
              pts = 0.0                                                         
          breakdown["rating_quality"] = round(pts, 2)                           
                                                                                
          # 2. Review volume (20 pts) — log scale                               
          if entity.review_count > 0:                                           
              pts = min(math.log10(entity.review_count + 1) / math.log10(5000), 
   1.0) * WEIGHTS["review_volume"]                                              
          else:                                                                 
              pts = 0.0                                                         
          breakdown["review_volume"] = round(pts, 2)                            
                                                                                
          # 3. Recency (20 pts)                                                 
          days = entity.days_since_last_review                                  
          if days <= 7: pts = 20.0                                              
          elif days <= 30: pts = 15.0                                           
          elif days <= 90: pts = 10.0                                           
          elif days <= 180: pts = 5.0                                           
          else: pts = 0.0                                                       
          breakdown["recency"] = pts                                            
                                                                                
          # 4. Price-value tier (15 pts) — mid-tier rewarded                    
          tier_pts = {1: 10.0, 2: 15.0, 3: 12.0, 4: 8.0}                        
          breakdown["price_value"] = tier_pts.get(entity.price_tier, 10.0)      
                                                                                
          # 5. Profile completeness (15 pts)                                    
          pts = 0.0                                                             
          if entity.has_photos: pts += 5.0                                      
          if entity.has_phone: pts += 5.0                                       
          if entity.has_website: pts += 5.0                                     
          breakdown["completeness"] = pts                                       
                                                                                
          total = sum(breakdown.values())                                       
          final_score = round(self.clamp(total), 2)                             
                                                                                
          return ScoreResult(                                                   
              entity_id=entity.id,                                              
              score=final_score,                                                
              breakdown=breakdown,                                              
          )
